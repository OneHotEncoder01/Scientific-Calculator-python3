# Menu

def Exit_m():
	Exit_m = tkinter.messagebox.askyesno("Scientific Calculator","confirm if you want to exit")
	if Exit_m > 0:
		root.destroy()
		return

def Scientific():
	scientific_calc.main()
	root.destroy()
	return


def Standart():
